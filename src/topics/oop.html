Prototypes

      
        function Person(name, age) {
          let person = Object.create(personPrototype);
          person.name = name;
          person.age = age;
          return person;
        }
        const personPrototype = {
          sayHi: () => {
            alert('Hi!');
          }
        };
        const personInstance = Person('Irene', 27);

Prototype chaining

      
        personPrototype.isPrototypeOf(personInstance); // true
        
        Object.prototype.isPrototypeOf(personPrototype); // true
        
        Object.prototype.isPrototypeOf(personInstance); // true

Objects

The new keyword

      
        function Person(name, age) {
          this.name = name;
          this.age = age;
        }
        Person.prototype.sayHi = () => {
          alert('Hi!');
        }
        const personInstance = new Person('Irene', 27);

The new keyword syntax is more widely used and accepted, so it's preferred over Object.create().

Classes

      
        class Animal {
          constructor(name, specie) {
              this.name = name;
              this.specie = specie;
          }
          walk() {
            return `${this.name} is walking.`;
          }
          jump() {
            return `${this.name} is jumping.`;
          }
        }
        const tomillo = new Animal('Tomillo', 'Cat');

This is syntatic sugar, since javascript is still using its prototypes underneath it.

Classes

Inheritance

      
        class Animal {
          constructor(name, age, specie) {
              this.name = name;
              this.age = age;
              this.specie = specie;
          }
          walk() {
            return `${this.name} is walking.`;
          }
        }

        class Cat extends Animal {
          constructor(name, age, color) {
            super(name, age, 'cat');
            this.color = color;
          }
          sleep() {
            return `${this.name} is sleeping.`;
          }
        }
        const tomillo = new Cat('Tomillo', 3, 'yellow');
